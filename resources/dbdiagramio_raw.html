Table body {

body_id varchar [pk] name varchar [not null] description varchar created datetime

}

Table minutes_item {

minutes_item_id varchar [pk] name varchar [not null] matter varchar legistar_event_item_id integer created datetime

}

Table minutes_item_file {

minutes_item_file_id varchar [pk] minutes_item_id varchar [ref: > minutes_item.minutes_item_id] uri varchar [not null] name varchar legistar_matter_attachment_id integer created datetime

}

Table event {

event_id varchar [pk] body_id varchar [ref: > body.body_id] agenda_file_uri varchar event_datetime varchar [not null] minutes_file_uri varchar source_uri varchar [not null] video_uri varchar [not null] legistar_event_id integer legistar_event_link varchar created datetime

}

Table event_minutes_item {

event_minutes_item_id varchar [pk] event_id varchar [ref: > event.event_id] minutes_item_id varchar [ref: > minutes_item.minutes_item_id] index integer decision varchar created datetime

}

Table person {

person_id varchar [pk] full_name varchar [not null] email varchar [not null] phone varchar website varchar legistar_person_id integer created datetime

}

Table vote {

vote_id varchar [pk] person_id varchar [ref: > person.person_id] event_minutes_item_id varchar [ref: > event_minutes_item.event_minutes_item_id] decision varchar [not null] legistar_event_item_vote_id integer created datetime

}

Table file {

file_id varchar [pk] uri varchar [not null] filename varchar description varchar content_type varchar created datetime

}

Table transcript {

transcript_id varchar [pk] event_id varchar [ref: > event.event_id] file_id varchar [ref: - file.file_id] confidence float [not null] created datetime

}

Table algorithm {

algorithm_id varchar [pk] name varchar [not null] version varchar [not null] description varchar source varchar created datetime

}

Table run {

run_id varchar [pk] algorithm_id varchar [ref: > algorithm.algorithm_id] begin datetime completed datetime

}

Table run_input {

run_input_id varchar [pk] run_id varchar [ref: > run.run_id] dtype varchar [not null] value varchar [not null]

}

Table run_input_file {

run_input_file_id varchar [pk] run_id varchar [ref: > run.run_id] file_id varchar [ref: - file.file_id]

}

Table run_output {

run_output_id varchar [pk] run_id varchar [ref: > run.run_id] dtype varchar [not null] value varchar [not null]

}

Table run_output_file {

run_output_file_id varchar [pk] run_id varchar [ref: > run.run_id] file_id varchar [ref: - file.file_id]

}

Table indexed_event_term {

indexed_event_term_id varchar [pk] term varchar [not null] event_id varchar [ref: > event.event_id] updated datetime

}

Table indexed_minutes_item_term {

indexed_minutes_item_term_id varchar [pk] term varchar [not null] minutes_item_id varchar [ref: > minutes_item.minutes_item_id] updated datetime

}
